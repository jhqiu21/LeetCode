class Solution {
    public int[] twoSum(int[] nums, int target) {
        int high = nums.length;
        HashMap<Integer, Integer> h = new HashMap<>();
        for (int i = 0; i < high; i++) {
            int val = target - nums[i];
            if (h.containsKey(val)) {
                return new int[]{i, h.get(val)};
            }
            h.put(nums[i], i);
        }
        return new int[2];
    }
}
